  lcd.begin(16, 2);

void loop() {

  float h = dht.readHumidity();          // read humidity

  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Temp.: ");                  // write temperature 
  lcd.print(t);
  lcd.print(" C");
  lcd.setCursor(0, 1);
  lcd.print("Humi.: ");                  // write humidity
  lcd.print(h);
  lcd.print("%");
  delay(1000);
}
